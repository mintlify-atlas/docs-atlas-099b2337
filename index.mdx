  <h2 className="text-2xl sm:text-3xl font-bold dark:text-white text-gray-900 mb-4">Quick start</h2>
  <p className="text-base dark:text-gray-400 text-gray-600 mb-8">Get up and running with React Native Date Picker in minutes</p>
  
  <Steps>
    <Step title="Install the package">
      Install react-native-date-picker using your preferred package manager.
      
      <CodeGroup>
      ```bash npm
      npm install react-native-date-picker
      ```
      ```bash yarn
      yarn add react-native-date-picker
      ```
      ```bash pnpm
      pnpm add react-native-date-picker
      ```
      </CodeGroup>
    </Step>
    
    <Step title="Install iOS dependencies">
      For non-Expo projects, install the CocoaPods dependencies.
      
      ```bash
      cd ios && pod install
      ```
      
      <Note>Expo projects can skip this step when using Development Builds.</Note>
    </Step>
    
    <Step title="Import and use the picker">
      Import DatePicker and add it to your component. Here's a modal example:
      
      ```jsx
      import React, { useState } from 'react'
      import { Button } from 'react-native'
      import DatePicker from 'react-native-date-picker'

      export default () => {
        const [date, setDate] = useState(new Date())
        const [open, setOpen] = useState(false)

        return (
          <>
            <Button title="Open" onPress={() => setOpen(true)} />
            <DatePicker
              modal
              open={open}
              date={date}
              onConfirm={(date) => {
                setOpen(false)
                setDate(date)
              }}
              onCancel={() => {
                setOpen(false)
              }}
            />
          </>
        )
      }
      ```
    </Step>
  </Steps>
</div>
